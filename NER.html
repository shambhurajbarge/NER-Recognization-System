from keras.models import Sequential
from keras.layers import Dense, Embedding, GlobalMaxPool1D, Dropout
from keras.optimizers import Adam

model = Sequential()
# embedding layer
model.add(Embedding(5000, 20, input_length=180))
model.add(Dropout(0.15))
# lstm layer
model.add(LSTM(20, return_sequences=False))
model.add(BatchNormalization())
model.add(Dropout(0.4))

# o/p layer-1
model.add(Dense(17, activation='softmax'))

from sklearn.metrics import confusion_matrix
#lets write a fun to find precision,recall,f1-score and accuracy by lablewise.
def print_results(headline, true_value, pred):
    print(headline)
    print("classification_report: {}".format(classification_report(true_value, pred)))
    conf = confusion_matrix(true_value, pred)
    conf.diagonal()/conf.sum(axis=1)
    print('accuracy by lable')
    print(conf.diagonal()/conf.sum(axis=1))
